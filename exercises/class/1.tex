	\begin{lstlisting}[language=C++]  		
#include <iostream>

class A;

class B {
	public:
	B() { std::cout << "B"; }
	friend B A::createB();
};

class A {
	public:
	A() { std::cout << "A"; }
	
	B createB() { return B(); }
};

int main() {
	A a;
	B b = a.createB();
}
